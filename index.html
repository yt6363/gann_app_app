// Initialize Firebase firebase.initializeApp(firebaseConfig); function signIn() { const provider = new firebase.auth.GoogleAuthProvider(); firebase.auth().signInWithPopup(provider) .then((result) => result.user.getIdToken()) .then((token) => { // Redirect to your Cloud Run app with the token as a query parameter window.location.href = "https://gann-app-142467784972.us-central1.run.app?token=" + token; }) .catch((error) => { console.error("Login error:", error); alert("Login failed: " + error.message); }); }
